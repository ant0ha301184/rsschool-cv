    import sys
    from tkinter.messagebox import *

        print("Передайте имя файла в качестве аргумента!")

    try:
        with open(sys.argv[1], "r") as file:
            data = file.readline()
                            
            while data != "":
                # data = data.rstrip()
                data = data.lower()
                for i in data:
                    if i == "\n":
                        continue
                    if i not in rezult.keys():
                        rezult[i] = 1
                    else:
                        rezult[i] += 1
                            
                data = file.readline()
    except IOError:
        showwarning("Ошибка при доступе к файлу!")
